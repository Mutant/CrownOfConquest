select * from Dungeon_Sector_Path dsp 
join Dungeon_Grid src on (src.dungeon_grid_id = dsp.sector_id) 
join Dungeon_Grid dst on (dst.dungeon_grid_id = dsp.has_path_to) 
where src.x=27 and src.y=17 and dst.x=27 and dst.y=19

SELECT me.sector_id, me.has_path_to, me.distance FROM Dungeon_Sector_Path me WHERE ( ( me.has_path_to = '33636' 
AND me.sector_id = '33643' ) )

SELECT me.sector_id, me.has_path_to, me.distance, doors_in_path.sector_id, doors_in_path.has_path_to, 
doors_in_path.door_id, door.door_id, door.position_id, door.dungeon_grid_id, door.type, door.state FROM Dungeon_Sector_Path me 
LEFT JOIN Dungeon_Sector_Path_Door doors_in_path ON ( doors_in_path.has_path_to = me.has_path_to 
AND doors_in_path.sector_id = me.sector_id ) 
LEFT JOIN Door door ON door.door_id = doors_in_path.door_id WHERE ( ( distance <= '3' AND me.sector_id = '33450' ) ) 
ORDER BY doors_in_path.has_path_to, doors_in_path.sector_id
